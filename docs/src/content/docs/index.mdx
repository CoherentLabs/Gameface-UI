        let index = 0;
        const step = 3;
        const codeElement = document.getElementById("code-block");

        function typeCode() {
            if (!window.Prism || !window.Prism.languages.jsx) {
                requestAnimationFrame(typeCode);
                return;
            }

            if (index < codeSnippet.length) {
                codeElement.textContent = codeSnippet.substring(0, index + step);
                Prism.highlightElement(codeElement);
                index+=step;
                requestAnimationFrame(typeCode);
            }
        }
        requestAnimationFrame(typeCode);
      })();
